	\umlclass[x=0, y=0,
	]{linear}{$\dot{\vec{x}} = \mat{A}\vec{x}$}{$\mat{A} \in \Real^{n \times n}$}

	\umlclass[x=-5, y=4,
	]{polynomial}{$\dot{\vec{x}} = \mathfrak{f}\!\left(\vec{x}\right)$}{$\mathfrak{f} \in \Poly[\Real]{\vec{x}}^n$}
	
	\umlclass[x=-5, y=8,
	]{rational}{$\dot{\vec{x}} = \mathfrak{G}\!\left(\vec{x}\right)^{-1} \mathfrak{f}\!\left(\vec{x}\right)$}{$\mathfrak{f} \in \Poly[\Real]{\vec{x}}^n$, $\mathfrak{G} \in \Poly[\Real]{\vec{x}}^{n \times n}$}
	
	\umlclass[x=5, y=4
	]{describing function}{$\dot{\vec{x}} = \mat{A} \vec\varphi\!\left(\vec{x}\right)$}{$\mat{A} \in \Real^{n \times n}$, $\vec\varphi\!: \Real^n \rightarrow \Real^n$}

	\umlclass[x=0, y=12,
	]{switching}{$\dot{\vec{x}} = \vec{h}_i\!\left(\vec{x}\right) \quad \text{if $\vec{x} \in \Phi_i$}$}{$\vec{h}\!: \mathcal{I} \times \Real^n \rightarrow \Real^n$ \\ $\Phi_i \subset \Real^n$}
	
	\umlclass[x=0, y=16,
	]{hybrid}{$\dot{\vec{x}} = \vec{h}_i\!\left(\vec{x}\right), \quad i = k_i\!\left(\vec{x}\right)$}{$\vec{h}\!: \mathcal{I} \times \Real^n \rightarrow \Real^n$ \\ $k\!: \mathcal{I} \times \Real^n \rightarrow \mathcal{I}$}
	
	\umlnote[x=2, y=9,
	]{switching}{$\left(\Phi_i\right)_{i \in \mathcal{I}}$ form a set partition of $\Real^n$.}

	\umlnote[x=-6, y=14]{hybrid}{}	
	\umlnote[x=-6, y=14,
	]{switching}{$\mathcal{I} \subset \Natural$ is the set of discrete {\em modes}.}
	
	
	\umlinherit{linear}{polynomial}
	\umlinherit{polynomial}{rational}
	\umlinherit{rational}{switching}
	\umlinherit{switching}{hybrid}
	
	\umlinherit{linear}{describing function}
	\umlimpl[geometry=|-]{describing function}{hybrid}
